% ================================================================ 1. MOTIVAÇÃO
\section{Motivação}

\begin{frame}{Ocultações estelares: medir o invisível}
  \begin{columns}[c]
    \column{0.52\textwidth}
    \begin{itemize}
      \item Um corpo do Sistema Solar cruza a frente de uma estrela.
      \item A \alert{curva de luz} revela tamanho, forma, anéis, atmosfera.
      \item Precisão de quilômetros a dezenas de UA.
    \end{itemize}
    \column{0.48\textwidth}
    \centering
    \includegraphics[width=\linewidth]{ocultacao_ilustracao.png}
  \end{columns}
  \vfill
  {\scriptsize Crédito: Lucie Maquet (2014) --- sistema de anéis de Chariklo.}
\end{frame}

\begin{frame}{Dois níveis de exigência}
  \begin{block}{O mínimo}
    Detectar a \textbf{queda profunda} do corpo principal --- sinal forte, fácil.
  \end{block}
  \begin{block}{O que importa de verdade}
    Não perder os \alert{sinais sutis} --- anéis, atmosferas tênues.
  \end{block}
  \vfill
  \centering
  Os \alert{anéis de Quaoar} foram descobertos só em uma \textbf{revisita} dos dados.\\
  {\footnotesize Um alerta automático nesses casos pode salvar uma descoberta.}
\end{frame}

% ================================================================ 2. PROBLEMA
\section{O problema}

\begin{frame}{O gargalo: \textit{big data} em ocultações}
  \begin{itemize}
    \item Redes de telescópios $+$ Gaia $\Rightarrow$ volume massivo de curvas.
    \item Triagem manual não escala.
    \item Ruído atmosférico/instrumental \textbf{imita} ocultações.
  \end{itemize}
  \vfill
  \begin{exampleblock}{Formulação}
    \centering
    \textbf{Classificação binária:} curva $\rightarrow$ vetor de \textit{features} $\rightarrow$
    \alert{positiva} (ocultação) ou \alert{negativa}.
  \end{exampleblock}
  \vfill
  {\footnotesize Abordagem: \textit{features} clássicas $+$ \textit{ensembles} de árvores --- interpretáveis, não caixa-preta.}
\end{frame}

\section{Estudo de caso: Quaoar}

\begin{frame}{Teste final: uma curva real de Quaoar}
  \begin{itemize}\footnotesize
    \item Dados \res{externos ao treino} (Gemini, CFHT) --- cedidos por Pereira et al.\ (2023).
    \item Corpo principal $+$ anéis Q1R e Q2R numa só série temporal.
  \end{itemize}
  \centering
  \includegraphics[width=0.9\linewidth]{quaoar_curva.png}
\end{frame}

\begin{frame}{Revisitar por recortes}
  \centering
  \includegraphics[width=0.8\linewidth]{quaoar_recortes.png}
  \vfill
  {\footnotesize O anel fino real recebe $\hat{p}\approx 4$--$8\%$ --- baixo em absoluto, mas muito acima do ruído vizinho.}
\end{frame}

\begin{frame}[standout]
  A separação é o que importa
  \vspace{1.2em}

  \bignum{$\sim$86$\times$}{probabilidade do anel real vs.\ ruído vizinho}
  \vspace{1em}

  \bignum{50\% $\rightarrow$ 100\%}{\textit{recall} sobre os eventos físicos, com $\tau=0{,}03$}
  \vspace{0.8em}

  {\footnotesize\itshape sem nenhum falso alarme nos recortes-controle}
\end{frame}

\begin{frame}{O que Quaoar demonstra}
  \begin{itemize}
    \item Não é o valor absoluto da probabilidade --- é a \alert{separação} ($\sim$2 ordens de grandeza).
    \item Demonstra, em dado real externo ao treino, a \textbf{revisita de curvas em busca de eventos sutis}.
    \item Janela mais larga \textbf{piora} (causa estrutural: \textit{features} globais) $\to$ trabalho futuro.
  \end{itemize}
\end{frame}

% ================================================================ 7. CONCLUSÕES
\section{Conclusões}

\begin{frame}{Conclusões}
  \begin{itemize}
    \item \textit{Pipeline} interpretável com \res{F1 $\approx$ 0,99}, robusta à simplificação de \textit{features}.
    \item Importância $\neq$ insubstituibilidade (K-means dispensável).
    \item Ajuste de $\tau$ incorpora o custo assimétrico \textbf{sem retreinar}.
    \item Em Quaoar: separa anel real de ruído por \res{$\sim$2 ordens de grandeza}.
  \end{itemize}
  \vfill
  \begin{block}{Trabalhos futuros}
    \footnotesize Janela deslizante e \textit{features} locais $\bullet$ \textit{holdout} 100\% real $\bullet$
    classificação multi-classe (corpo, anel, atmosfera).
  \end{block}
\end{frame}
